<mx:HBox xmlns:mx="http://www.adobe.com/2006/mxml" width="100%"   verticalAlign="middle" paddingLeft="5" borderStyle="solid" borderColor="#0000a0" cornerRadius="4" themeColor="#0000ff" borderThickness="1">
<mx:Metadata>
    [Event(type="movieSelectedEvent", name="titleClicked")]
    [Event(type="movieSelectedEvent", name="showComments")]
</mx:Metadata>
<mx:Script>
    <![CDATA[
        import mx.controls.Image;
        import com.rbm.valueObjects.MovieItem;
        import mx.core.DragSource;
        import mx.managers.DragManager;
        import mx.events.DragEvent;
        import mx.core.IUIComponent;
        import events.MovieSelectedEvent;
        
        [Bindable]
        public var myItem:MovieItem = null;
        
        private function dragMe(initiator:Image, dsData:MovieItem , event:MouseEvent, format:String):void{
            var ds:DragSource = new DragSource();
            ds.addData(dsData,format);
            DragManager.doDrag(initiator,ds,event);
        }
        
        private function titleClicked():void{
            var e:MovieSelectedEvent = new MovieSelectedEvent(MovieItem(myItem) , 'titleClicked');
            //trace(this.myItem.moviePath);
            this.dispatchEvent(e);
            
        }
        
        
        private function commentsClicked():void{
            var e:MovieSelectedEvent = new MovieSelectedEvent(MovieItem(myItem) , 'showComments');
            //trace(this.myItem.moviePath);
            this.dispatchEvent(e);
            
        }
    ]]>
</mx:Script>
    <mx:Image id="movieTeaserImage" styleName="movieTeaserImage" source="{'assets/images/'+myItem.teaserImage}"  height="64" width="50" mouseDown="dragMe(movieTeaserImage, myItem ,event,'movieItemFormat')"  themeColor="#00ff00" scaleContent="true"/>
    <mx:VBox width="100%" verticalGap="0">
        <mx:Label id="movieName" text="{myItem.movieName}" styleName="movieTitle" click="titleClicked()" />
        <mx:Label id="viewComments" text="View Comments"  click="commentsClicked()" />
        <mx:Label id="totalPlays" text="Total Plays: {myItem.totalPlays}" styleName = "movieTotalPlays" />
        <mx:TextArea id="movieDescription" text="{myItem.movieDescription}" styleName="movieDescription"  backgroundAlpha="0.0" borderStyle="none" wordWrap="true"  editable="false"  width="100%"   height="100%"/>
    </mx:VBox>
</mx:HBox>
