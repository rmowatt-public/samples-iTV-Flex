<?xml version="1.0" encoding="utf-8"?>
<mx:Panel xmlns:mx="http://www.adobe.com/2006/mxml" styleName="ColorController" width="150" height="{(mama.height * 1.1) + this.getHeaderHeight()}" title="Colors" >
    <mx:Metadata>
        [Event(name="colorUpdated", type="events.ColorChangeEvent")]
    </mx:Metadata>
    <mx:VBox  backgroundColor="white" width="99%" id="mama" paddingBottom="3" >
        <mx:Script>
            <![CDATA[
                import events.ColorChangeEvent;
                import mx.events.SliderEvent;
                import mx.controls.sliderClasses.Slider;
                import com.rbm.valueObjects.Color.Color;
                
                private function slideMe(event:SliderEvent):void{
                    var sliders:Array = mama.getChildren();
                    var colors:Array = new Array();
                    for each(var s:Object in sliders){
                        if(s.id != 'btn'){
                                colors[s.id] = Math.round((Number(Slider(s).value.toPrecision(2))-5)*20);
                        }
                    }
                    var clr:Color = new Color(colors['red'],colors['blue'],colors['green']);
                    var e:ColorChangeEvent = new ColorChangeEvent(clr, 'colorUpdated');
                    this.dispatchEvent(e);
                    //trace( clr.toString());
                }
                
                private function setThumbValue(slide:HSlider , num:Number):void{
                    slide.setThumbValueAt(0,num);
                }
                
                private function resetThumbVals():void{
                    var sliders:Array = mama.getChildren();
                    for each(var s:Object in sliders){;
                        if(s.id != 'btn'){
                        HSlider(s).setThumbValueAt(0,5);
                        }
                    }
                    
                }
            ]]>
        </mx:Script>
        <mx:HSlider id="red" change="slideMe(event)" liveDragging="true" tickInterval="2" labels="['min', 'max']" labelStyleName="sliderLabel" width="{mama.width}" 
                      fillColors="[0xFF0000,0xFF0000,0xFF0000,0xFF0000;]"
                     creationComplete="setThumbValue(this.red,5)"/>
        <mx:HSlider id="blue" change="slideMe(event)" liveDragging="true" tickInterval="2" labelStyleName="sliderLabel" width="{mama.width}"
                    fillColors="[0x0000FF, 0xCCCCCC, 0xEEFFFF, 0xEE0000;]" 
                    creationComplete="setThumbValue(this.blue,5)"/>
        <mx:HSlider id="green" change="slideMe(event)" liveDragging="true" tickInterval="2"  labelStyleName="sliderLabel"  width="{mama.width}"
                    creationComplete="setThumbValue(this.green,5)"/>
        <mx:HBox  id='btn' width="100%" horizontalAlign="right">
            <mx:Button  label="reset" click="resetThumbVals()"  height="13" styleName="colorReset"/>
        </mx:HBox>
    </mx:VBox>
</mx:Panel>
