<mx:Panel xmlns:mx="http://www.adobe.com/2006/mxml" layout="horizontal" xmlns:c="com.rbm.displayObjects.CommentPanel.*" 
            xmlns:vs="com.rbm.displayObjects.VideoSelection.*" title="Other Videos" creationComplete="movieInventory.send()">
    <mx:Script>
        <![CDATA[
            import com.rbm.displayObjects.VideoSelection.MovieItemObject;
            import com.rbm.valueObjects.*;
            import mx.collections.ArrayCollection;
            import mx.rpc.events.ResultEvent;
            
            [Bindable]
            public var myMovie:MovieItem;
            
            [Bindable]
            public var requestURL:String;
            
            private function makeMovies(event:ResultEvent):void{
                for each(var m:XML in event.result..movie){
                    myMovie = new MovieItem(12 , m.movieName , m.moviePath , m.movieDescription, m.totalPlays, m.teaserImage);
                    var vs:MovieItemObject= new MovieItemObject();
                    vs.label = m.movieName;
                    vs.myItem = myMovie;
                    trace(m.moviePath);
                this.mAccordion.addChild(vs);
                }
            }
        ]]>
    </mx:Script>
    <mx:HTTPService id="movieInventory" url="{requestURL}" result="makeMovies(event)" resultFormat="e4x"/>
    <mx:Accordion selectedIndex="3" id="mAccordion" height="100%" width="100%" />
</mx:Panel>
